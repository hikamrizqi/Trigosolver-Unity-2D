   void Start() {
      HideAllGameObjects(false);  // Hide tanpa animasi
      ShowLevelPanel();            // Panel slide in
   }
   
   void OnLevelSelected(int level) {
      StartCoroutine(LevelSelectionSequence(level));
   }
   
   IEnumerator LevelSelectionSequence(int level) {
      // 1. Panel slide out
      levelSelectionPanel.transform.DOMoveY(
         levelSelectionPanel.transform.position.y + slideDistance, 
         animationDuration
      ).SetEase(slideEase);
      
      yield return new WaitForSeconds(animationDuration);
      levelSelectionPanel.SetActive(false);
      
      // 2. Game objects slide in
      ShowAllGameObjects(true);
      
      // 3. Start game
      int questionNumber = (level == 1) ? 1 : 11;
      calculationManager.StartFromQuestion(questionNumber);
   }
